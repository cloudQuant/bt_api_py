### 1.2 新流程设计

```python
# bt_api.py 新启动流程
from bt_api_base.plugins.loader import PluginLoader
from bt_api_base.registry import ExchangeRegistry

class BtApi:
    def __init__(self, exchange_config=None, *, zmq_config=None, auto_load_plugins=True):
        self._registry = ExchangeRegistry()

        # 通过 entry_points 自动发现已安装的 bt_api_xx 插件
        if auto_load_plugins:
            loader = PluginLoader(self._registry)
            loaded = loader.load_all()
            self.logger.info(f"Loaded {len(loaded)} plugins: {[p.name for p in loaded]}")

        # ZMQ 分发（可选）
        self._zmq_publisher = None
        self._zmq_server = None
        if zmq_config:
            self._init_zmq(zmq_config)

        # 初始化交易所连接
        if exchange_config:
            for exchange_name, params in exchange_config.items():
                self.add_exchange(exchange_name, **params)
```


### 2.3 Topic 格式规范

```python
@dataclass(frozen=True)
class Topic:
    exchange: str       # BINANCE, OKX, CTP
    asset_type: str     # SPOT, SWAP, FUTURES, OPTION
    symbol: str         # BTC-USDT, ETH-USDT
    data_type: str      # ticker, depth, kline, order, trade, balance, position

    def __str__(self) -> str:
        return f"{self.exchange}.{self.asset_type}.{self.symbol}.{self.data_type}"

    @classmethod
    def from_str(cls, topic_str: str) -> "Topic":
        parts = topic_str.split(".")
        if len(parts) != 4:
            raise ValueError(f"Invalid topic: {topic_str}")
        return cls(*parts)

    @classmethod
    def from_dataname(cls, dataname: str, data_type: str) -> "Topic":
        """从 BtApi 的 dataname (BINANCE___SPOT___BTC-USDT) 转换"""
        parts = dataname.split("___")
        return cls(exchange=parts[0], asset_type=parts[1], symbol=parts[2], data_type=data_type)
```

### 2.5 ZmqPublisher 核心逻辑

```python
class ZmqPublisher:
    def __init__(self, config: ZmqConfig):
        self._ctx = zmq.asyncio.Context(io_threads=config.io_threads)
        self._socket = self._ctx.socket(zmq.PUB)
        self._socket.setsockopt(zmq.SNDHWM, config.hwm)
        self._socket.setsockopt(zmq.LINGER, config.linger)
        self._socket.bind(config.pub_addr)
        self._serializer = get_serializer(config.serializer)
        self._sequence = 0

    async def publish(self, topic: Topic | str, data: dict) -> None:
        self._sequence += 1
        msg = Message(
            topic=str(topic),
            timestamp=int(time.time() * 1000),
            data=data,
            sequence=self._sequence,
        )
        payload = self._serializer.serialize(msg)
        # 多帧消息：topic (用于过滤) + payload (实际数据)
        await self._socket.send_multipart([
            str(topic).encode(),
            payload,
        ])

    async def heartbeat_loop(self, interval: float = 5.0) -> None:
        while True:
            await self.publish("_heartbeat", {"ts": time.time()})
            await asyncio.sleep(interval)
```

### 2.6 ZmqSubscriber 核心逻辑

```python
class ZmqSubscriber:
    def __init__(self, address: str, serializer: str = "msgpack"):
        self._ctx = zmq.asyncio.Context()
        self._socket = self._ctx.socket(zmq.SUB)
        self._socket.connect(address)
        self._serializer = get_serializer(serializer)
        self._callbacks: dict[str, list[Callable]] = defaultdict(list)
        self._last_heartbeat = time.time()

    def subscribe(self, topic_pattern: str, callback: Callable) -> None:
        """订阅指定 topic 模式（前缀匹配）"""
        self._socket.setsockopt_string(zmq.SUBSCRIBE, topic_pattern)
        self._callbacks[topic_pattern].append(callback)

    async def run(self) -> None:
        """主循环：接收消息并分发到 callback"""
        while True:
            topic_bytes, payload = await self._socket.recv_multipart()
            topic_str = topic_bytes.decode()
            msg = self._serializer.deserialize(payload)

            if topic_str == "_heartbeat":
                self._last_heartbeat = time.time()
                continue

            for pattern, callbacks in self._callbacks.items():
                if topic_str.startswith(pattern):
                    for cb in callbacks:
                        try:
                            if asyncio.iscoroutinefunction(cb):
                                await cb(msg)
                            else:
                                cb(msg)
                        except Exception as e:
                            logger.error(f"Callback error: {e}")
```

### 3.2 接口定义

```python
from abc import ABC, abstractmethod
from typing import Any
from collections.abc import Callable


class AbstractDirectClient(ABC):
    """交易所直连客户端基类

    每个 bt_api_xx 包必须实现此基类，提供直连交易所的能力。
    所有方法均为 async，支持 async context manager。
    """

    def __init__(self, **kwargs):
        self._connected = False

    # ─── 生命周期 ───

    @abstractmethod
    async def connect(self) -> None:
        """建立连接"""
        ...

    @abstractmethod
    async def disconnect(self) -> None:
        """断开连接"""
        ...

    async def __aenter__(self):
        await self.connect()
        return self

    async def __aexit__(self, *args):
        await self.disconnect()

    # ─── 行情 REST ───

    @abstractmethod
    async def get_ticker(self, symbol: str) -> dict[str, Any]:
        """获取最新行情"""
        ...

    @abstractmethod
    async def get_depth(self, symbol: str, limit: int = 20) -> dict[str, Any]:
        """获取深度数据"""
        ...

    @abstractmethod
    async def get_kline(self, symbol: str, period: str,
                        limit: int = 100) -> list[dict[str, Any]]:
        """获取K线数据"""
        ...

    # ─── 交易 ───

    @abstractmethod
    async def place_order(self, symbol: str, side: str, order_type: str,
                          price: float | None, quantity: float,
                          **kwargs) -> dict[str, Any]:
        """下单"""
        ...

    @abstractmethod
    async def cancel_order(self, symbol: str, order_id: str) -> bool:
        """撤单"""
        ...

    @abstractmethod
    async def get_order(self, symbol: str, order_id: str) -> dict[str, Any]:
        """查询订单"""
        ...

    @abstractmethod
    async def get_open_orders(self, symbol: str | None = None) -> list[dict[str, Any]]:
        """查询未完成订单"""
        ...

    # ─── 账户 ───

    @abstractmethod
    async def get_balance(self) -> list[dict[str, Any]]:
        """查询余额"""
        ...

    # ─── WebSocket 订阅 ───

    async def subscribe_ticker(self, symbol: str, callback: Callable) -> str:
        """订阅行情（可选实现）"""
        raise NotImplementedError

    async def subscribe_depth(self, symbol: str, callback: Callable) -> str:
        """订阅深度（可选实现）"""
        raise NotImplementedError

    async def unsubscribe(self, subscription_id: str) -> bool:
        """取消订阅（可选实现）"""
        raise NotImplementedError
```

### 3.3 BinanceDirectClient 示例结构

```python
# bt_api_binance/src/bt_api_binance/client.py

class BinanceDirectClient(AbstractDirectClient):
    def __init__(self, api_key: str = "", secret: str = "",
                 testnet: bool = False, asset_type: str = "SPOT"):
        super().__init__()
        self._api_key = api_key
        self._secret = secret
        self._base_url = (
            "https://testnet.binance.vision" if testnet
            else "https://api.binance.com"
        )
        self._session: aiohttp.ClientSession | None = None

    async def connect(self) -> None:
        self._session = aiohttp.ClientSession()
        self._connected = True

    async def disconnect(self) -> None:
        if self._session:
            await self._session.close()
        self._connected = False

    async def get_ticker(self, symbol: str) -> dict:
        venue_symbol = symbol.replace("-", "")  # BTC-USDT → BTCUSDT
        resp = await self._request("GET", "/api/v3/ticker/24hr",
                                   params={"symbol": venue_symbol})
        return resp

    async def place_order(self, symbol: str, side: str, order_type: str,
                          price: float | None, quantity: float, **kwargs) -> dict:
        params = {
            "symbol": symbol.replace("-", ""),
            "side": side.upper(),
            "type": order_type.upper(),
            "quantity": str(quantity),
        }
        if price is not None:
            params["price"] = str(price)
            params["timeInForce"] = kwargs.get("time_in_force", "GTC")
        return await self._signed_request("POST", "/api/v3/order", params=params)
```

## 5. 数据流示意

### 5.1 行情数据流（ZMQ 模式）

```
交易所 WebSocket
    │
    ▼
bt_api_xx (plugin 中的 feed)
    │ callback / queue
    ▼
BtApi (EventBus.emit("ticker", data))
    │
    ├──▶ 本地策略回调 (同进程)
    │
    └──▶ ZmqPublisher.publish(topic, data)
            │
            ├──▶ ZmqSubscriber A (策略进程1)
            ├──▶ ZmqSubscriber B (策略进程2)
            └──▶ ZmqSubscriber C (策略进程3)
```

### 5.2 交易指令流（ZMQ 模式）

```
策略进程
    │
    ▼
ZmqTradeClient.place_order(...)
    │ REQ
    ▼
ZmqTradeServer (BtApi 侧)
    │ 解析请求
    ▼
BtApi.make_order(...)
    │
    ▼
bt_api_xx (plugin 中的 feed)
    │
    ▼
交易所 REST API
    │ 响应
    ▼
BtApi → ZmqTradeServer → ZmqTradeClient
    │ REP
    ▼
策略进程收到下单结果
```
